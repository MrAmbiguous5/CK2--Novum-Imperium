#Stolen Goods - Martial Mission Start
society_quest_event = {
	id = RTPmsociety.1
	desc = "RTPmsociety.1_desc"
	is_triggered_only = yes

	immediate = {
		add_character_modifier = {
			name = quest_cooldown_timer
			hidden = yes
			days = 1825
		}

		liege = {
			create_character = {
				age = 35
				random_traits = yes
				religion = PREV
				culture = PREV
				dynasty = none
				health = 10
			}
			new_character = {
				set_character_flag = rtp_trash #Just delete this dude
				save_event_target_as = rtp_society_quest_target #The first target of the quest
			}
		}

		society = {
			random_society_member = {
				limit = {
					rtp_is_capable_adult = yes
				}
				preferred_limit = {
					ROOT = {
						top_liege = {
							is_within_diplo_range = PREVPREV
						}
					}
				}
				save_event_target_as = rtp_society_quest_target2 #The second target of the quest
			}
		}
	}

	option = {
		name = "RTPmsociety.1_1"

		add_trait = travelling

		hidden_tooltip = {
			event_target:rtp_society_quest_target = {
				show_scope_change = no
				set_quest_target = {
					id = rtp_stolen_goods_martial_mission
					holder = ROOT
				}
			}
		}

		character_event = {
			id = RTPmsociety.2 #Stolen Goods - Martial Mission Part 1
			days = 90
		}
	}

	option = {
		name = "RTPmsociety.1_2"

		custom_tooltip = {
			text = "rtp_may_be_kicked_ct"

			random = {
				chance = 25
	
				society = {
					leader = {
						ROOT = {
							letter_event = {
								id = RTPsociety.2 #Society Member Gets Kicked Out From The Society
							}
						}
					}
				}
			}
		}
	}
}
