#Ping Settling Iceland - Start
character_event = {
	id = RTPdestiny.105
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		year = 800
		year <= 850
		NOT = {
			has_global_flag = rtp_iceland_settled
		}
	}

	immediate = {
		set_global_flag = rtp_iceland_settled

		if = {
			limit = {
				any_player = {
					has_landed_title = c_landless_t
					OR = {
						religion = norse_pagan
						religion = norse_pagan_reformed
						culture = norse
					}
				}
			}
			any_player = {
				save_event_target_as = rtp_iceland_settler1 #You are the first settler
			}
		}

		random_independent_ruler = {
			limit = {
				OR = {
					religion = norse_pagan
					religion = norse_pagan_reformed
				}
			}
			preferred_limit = {
				culture = norse
			}
			create_character = {
				age = 20
				random_traits = yes
				culture = norse
				religion = norse_pagan
				dynasty = random
			}
			new_character = {
				save_event_target_as = rtp_iceland_ai_settler1 #You are the first settler
			}
			create_character = {
				age = 21
				random_traits = yes
				culture = norse
				religion = norse_pagan
				dynasty = random
			}
			new_character = {
				save_event_target_as = rtp_iceland_ai_settler2 #You are the second settler
			}
			create_character = {
				age = 22
				random_traits = yes
				culture = norse
				religion = norse_pagan
				dynasty = random
			}
			new_character = {
				save_event_target_as = rtp_iceland_ai_settler3 #You are the third settler
			}
			create_character = {
				age = 23
				random_traits = yes
				culture = norse
				religion = norse_pagan
				dynasty = random
			}
			new_character = {
				save_event_target_as = rtp_iceland_ai_settler4 #You are the fourth settler
			}
		}

		any_player = {
			narrative_event = {
				id = RTPdestiny.6
			}
		}
	}
}

	id = RTPdestiny.1000
	hide_window = yes
	is_triggered_only = yes
	ai = no

	trigger = {
		year < 850
		NOT = {
			has_global_flag = rtp_iceland_settled
		}
	}

	immediate = {
		any_title = {
			limit = {
				OR = {
					title = c_austisland
					title = c_sunnlendingafjordungur
					title = c_nordlendingafjordungur
					title = c_vestisland
				}
			}
			location = {
				any_province_holding = {
					destroy_settlement = THIS
				}
			}
		}
	}
}
