	title = "RTPglobal.1_title"
	desc = "RTPglobal.1_desc"
	picture = GFX_evt_doge_republic

	only_playable = yes

		NOT = {
			has_character_flag = rtp_republic_switch #Tried switching already

			factor = 0
		}
