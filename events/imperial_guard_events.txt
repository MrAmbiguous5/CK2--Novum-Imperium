character_event = {
	id = toe.1300
	desc = EVTDESC_toe.1300
	picture = "GFX_evt_teaching_moment"
	is_triggered_only = yes
	#Removed trigger clause, if there are errors try putting it back
	immediate = {
		primary_title = {
			save_event_target_as = guard_parent

	}
	
	option = {
		name = EVTOPTA_toe.1300 #Local
		set_character_flag = chose_local
		hidden_tooltip = {
			if = {
				limit = {
					gender_equality_trigger = yes
				}
				create_character = {
					random_traits = yes
					religion = ROOT
					culture = ROOT
					dynasty = NONE
					female = 50
					age = 34
					trait = skilled_tactician
				}
			}
			else = {
				create_character = {
					random_traits = yes
					religion = ROOT
					culture = ROOT
					dynasty = NONE
					female = no
					age = 34
					trait = skilled_tactician
				}
			}
			new_character = {
				add_trait = toe_guard_captain
				save_event_target_as = guard_captain
				ROOT = { narrative_event = { id = toe.1302 }}
			}
		}
		capital_scope = {
			create_title = {
				tier = DUKE
				landless = yes
				name = LOCALGUARD #i.e. Byzantine Guard
				custom_created = yes
				base_title = PREV
				mercenary = yes
				army_template = livonian_order_composition
				ruler = "CAPTAIN"
				foa = "CAPTAIN_FOA"
				ruler_female = "CAPTAIN_FEMALE"
				holder = event_target:guard_captain
			}
		}
		event_target:guard_captain = {
			set_defacto_liege = ROOT
			primary_title = {
				set_title_flag = is_imperial_guard
				
				#Change guard sizes
				if = {
					limit = {
						year = 900
					}
					change_title_army_size = 0.5
				}
				if = {
					limit = {
						year = 1000
					}
					change_title_army_size = 0.75
				}
				if = {
					limit = {
						year = 1100
					}
					change_title_army_size = 1.13
				}
				if = {
					limit = {
						year = 1200
					}
					change_title_army_size = 1.69
				}
				if = {
					limit = {
						year = 1300
					}
					change_title_army_size = 2.53
				}
				if = {
					limit = {
						year = 1400
					}
					change_title_army_size = 3.8
				}
			}
			add_character_modifier = { 
				name = guard_captain
				duration = -1
			}
		}
		primary_title = {
			set_title_flag = local_guard
		}
	}
	option = {
		name = EVTOPTB_toe.1300 #Mixed
		trigger = {
			any_playable_ruler = {
				OR = {
					religion = norse_pagan
					religion = tengri_pagan
					religion = baltic_pagan
					religion = finnish_pagan
					religion = slavic_pagan
					religion = zun_pagan
					religion = west_african_pagan
					religion = aztec
					religion = bon
				}
				NOT = { religion = ROOT }
				capital_scope = {
					distance = {
						where = ROOT
						value = 400	# Not too far away
					}
				}
			}
		}
		set_character_flag = chose_local
		character_event = { id = toe.1301 }
		primary_title = {
			set_title_flag = local_guard
		}
	}
	option = {
		name = EVTOPTC_toe.1300 #Foreign
