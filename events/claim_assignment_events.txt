namespace = toe

#Give claims to council members

character_event = {
	id = toe.1100
	
	hide_window = yes

	trigger = {
		liege = {
			primary_title = {
				tier = EMPEROR
				temporary = no
				is_tribal = no
				is_nomadic = no
			}
			is_nomadic = no
			is_tribal = no
		}
		is_councillor = yes
	}
	
	mean_time_to_happen = { days = 1 }
	
	immediate = {
		liege = {
			primary_title = { 
				add_claim = ROOT
			}
		}
	}
}

#Give claims to husbands of princesses

character_event = {
	id = toe.1101
	
	hide_window = yes

	trigger = {
		liege = {
			primary_title = {
				tier = EMPEROR
				temporary = no
				is_tribal = no
				is_nomadic = no
				NOT = { has_law = succ_feudal_elective }
				NOT = { has_law = succ_elective_gavelkind }
				NOT = { has_law = succ_hre_elective }
				NOT = { has_law = succ_tanistry }
			}
			is_nomadic = no
			is_tribal = no
		}
		has_concubinage = no
		has_polygamy = no
		NOT = { has_religion_feature = religion_holy_family }
		NOT = { has_religion_feature = religion_harems }
		any_spouse = {
			is_close_relative = liege
		}
		
	}
	
	mean_time_to_happen = { days = 1 }
	
	immediate = {
		if = {
			limit = {
				OR = {
					AND = {
						has_religion_feature = religion_patriarchal
						is_female = no
					}
					AND = {
						has_religion_feature = religion_matriarchal
						is_female = yes
					}
				}
			}
			liege = {
				primary_title = { 
					add_claim = ROOT
				}
			}
		}
		if = {
			limit = {
				OR = {
					AND = {
						has_religion_feature = religion_patriarchal
						is_female = yes
					}
					AND = {
						has_religion_feature = religion_matriarchal
						is_female = no
					}
					has_religion_feature = religion_equal
				}
			}
			liege = {
				primary_title = { 
					add_claim = ROOT
				}
			}
		}
	}
}
