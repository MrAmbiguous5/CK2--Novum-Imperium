#Realm with designated regent inherited by realm that already has a designated regent
character_event = {
	id = lux_regency.9
	hide_window = yes
	
	is_triggered_only = yes
	
	only_rulers = yes
	
	trigger = {
		OR = {
			any_courtier = {
				has_minor_title = title_appointed_regent
			}
			any_vassal = {
				has_minor_title = title_appointed_regent
			}
		}
		current_heir = {
			is_ruler = yes
			OR = {
				any_courtier = {
					has_minor_title = title_appointed_regent
				}
				any_vassal = {
					has_minor_title = title_appointed_regent
				}
			}
		}
	}
	immediate = {
		random_courtier = {
			limit = {
				has_minor_title = title_appointed_regent
			}
			remove_title = title_appointed_regent
		}
		random_vassal = {
			limit = {
				has_minor_title = title_appointed_regent
			}
			remove_title = title_appointed_regent
		}
	}
}
