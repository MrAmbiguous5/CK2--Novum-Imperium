	has_character_flag = lent
	
	trigger = {
		OR = {
			age = 50
			is_weak_trigger = yes
		}
		NOT = { has_character_flag = lent_ill }
	}
	
	mean_time_to_happen = {
		days = 30
	}
	
	option = {
		name = "EVTOPTAlent.4"
		if = {
			limit = {
				NOT = {
					has_dlc = "Reapers"
				}
			}
			add_trait = ill
		}
		if = {
			limit = {
				has_dlc = "Reapers"
			}
			add_symptom_effect = yes
		}
		narrative_event = { id = lent.22 days = 1 tooltip = "EVTTOOLTIPlent.22" }
	}
	option = {
		name = "EVTOPTBlent.4"
		piety = 15
		set_character_flag = lent_ill
		if = {
			limit = {
				NOT = {
					has_dlc = "Reapers"
				}
			}
			add_trait = ill
			break = yes
		}
		add_symptom_effect = yes
	}
}
