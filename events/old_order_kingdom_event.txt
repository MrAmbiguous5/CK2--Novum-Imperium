namespace = OLDORDER # latest 0107

character_event = {
    id = OLDORDER.0020
    
	hide_window = yes
	
	trigger = {
		OR = {
			primary_title = {
				NOT = { government = order_government }
				NOT = { has_law = minor_holy_order_elective }
				has_title_flag = minor_holy_order
			}
			any_demesne_title = {
				NOT = { has_law = minor_holy_order_elective }
				has_title_flag = minor_holy_order
			}
		}
	}
	
	immediate = {
		primary_title = {
			if = {
				limit = {
					has_title_flag = minor_holy_order
				}
				set_goverment_type = order_government
				add_law = minor_holy_order_elective
			}
		}
		any_demesne_title = {
			if = {
				limit = {
					has_title_flag = minor_holy_order
				}
				add_law = minor_holy_order_elective
			}
		}
	}
}
